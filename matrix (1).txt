#include <stdio.h>

void inputMatrix(int a[10][10], int r, int c);
void displayMatrix(int a[10][10], int r, int c);
void addMatrix(int a[10][10], int b[10][10], int r, int c);
void multiplyMatrix(int a[10][10], int b[10][10], int r1, int c1, int c2);
void transposeMatrix(int a[10][10], int r, int c);

int main()
{
    int A[10][10], B[10][10];
    int r1, c1, r2, c2;

    printf("Enter rows and columns of Matrix A: ");
    scanf("%d%d", &r1, &c1);

    printf("Enter rows and columns of Matrix B: ");
    scanf("%d%d", &r2, &c2);

    printf("\nEnter elements of Matrix A:\n");
    inputMatrix(A, r1, c1);

    printf("\nEnter elements of Matrix B:\n");
    inputMatrix(B, r2, c2);

    printf("\nMatrix A:\n");
    displayMatrix(A, r1, c1);

    printf("\nMatrix B:\n");
    displayMatrix(B, r2, c2);

    // Matrix Addition
    if (r1 == r2 && c1 == c2)
    {
        printf("\nMatrix Addition:\n");
        addMatrix(A, B, r1, c1);
    }
    else
    {
        printf("\nMatrix Addition is not possible.\n");
    }

    // Matrix Multiplication
    if (c1 == r2)
    {
        printf("\nMatrix Multiplication:\n");
        multiplyMatrix(A, B, r1, c1, c2);
    }
    else
    {
        printf("\nMatrix Multiplication is not possible.\n");
    }

    // Transpose of Matrix A
    printf("\nTranspose of Matrix A:\n");
    transposeMatrix(A, r1, c1);

    return 0;
}

// Function to input matrix
void inputMatrix(int a[10][10], int r, int c)
{
    int i, j;
    for (i = 0; i < r; i++)
    {
        for (j = 0; j < c; j++)
        {
            scanf("%d", &a[i][j]);
        }
    }
}

// Function to display matrix
void displayMatrix(int a[10][10], int r, int c)
{
    int i, j;
    for (i = 0; i < r; i++)
    {
        for (j = 0; j < c; j++)
        {
            printf("%4d", a[i][j]);
        }
        printf("\n");
    }
}

// Function for matrix addition
void addMatrix(int a[10][10], int b[10][10], int r, int c)
{
    int i, j, sum[10][10];

    for (i = 0; i < r; i++)
    {
        for (j = 0; j < c; j++)
        {
            sum[i][j] = a[i][j] + b[i][j];
        }
    }

    displayMatrix(sum, r, c);
}

// Function for matrix multiplication
void multiplyMatrix(int a[10][10], int b[10][10], int r1, int c1, int c2)
{
    int i, j, k;
    int product[10][10];

    for (i = 0; i < r1; i++)
    {
        for (j = 0; j < c2; j++)
        {
            product[i][j] = 0;
            for (k = 0; k < c1; k++)
            {
                product[i][j] += a[i][k] * b[k][j];
            }
        }
    }

    displayMatrix(product, r1, c2);
}

// Function for transpose
void transposeMatrix(int a[10][10], int r, int c)
{
    int i, j;
    int transpose[10][10];

    for (i = 0; i < r; i++)
    {
        for (j = 0; j < c; j++)
        {
            transpose[j][i] = a[i][j];
        }
    }

    displayMatrix(transpose, c, r);
}
